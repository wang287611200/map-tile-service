## 5.2 完整数据库 Schema（PostgreSQL）

### 枚举类型

```sql
-- 数据类型
CREATE TYPE data_type_enum AS ENUM (
    'road_network',      -- 路网
    'poi',               -- 兴趣点
    'building',          -- 建筑物
    'terrain',           -- 地形
    'satellite'          -- 卫星影像
);

-- 版本状态
CREATE TYPE version_status AS ENUM (
    'DRAFT',
    'TESTING',
    'CANARY',
    'PUBLISHED',
    'DEPRECATED',
    'ROLLED_BACK'
);

-- 离线包类型
CREATE TYPE package_type AS ENUM (
    'full',         -- 全量包
    'incremental'   -- 增量包
);
```

### regions 表

```sql
CREATE TABLE regions (
    region_id       VARCHAR(32) PRIMARY KEY,     -- 区域唯一标识，如 CN_BJ
    region_name     VARCHAR(128) NOT NULL,        -- 区域中文名，如 北京市
    parent_region   VARCHAR(32) REFERENCES regions(region_id),
    bounding_box    JSONB NOT NULL,               -- {"west":115.4,"east":117.5,"south":39.4,"north":41.1}
    min_zoom        SMALLINT NOT NULL DEFAULT 3,
    max_zoom        SMALLINT NOT NULL DEFAULT 17,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_regions_parent ON regions(parent_region);
```

### data_format_versions 表

```sql
CREATE TABLE data_format_versions (
    format_version  VARCHAR(16) PRIMARY KEY,      -- 如 format_v3.0
    data_type       data_type_enum NOT NULL,
    description     TEXT,
    schema_spec_url VARCHAR(512),                 -- 格式规范文档地址
    breaking_change BOOLEAN NOT NULL DEFAULT FALSE,
    released_at     TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

### engine_versions 表

```sql
CREATE TABLE engine_versions (
    engine_version      VARCHAR(16) PRIMARY KEY,  -- 如 3.5.0
    release_date        DATE NOT NULL,
    min_data_format     VARCHAR(16) NOT NULL REFERENCES data_format_versions(format_version),
    max_data_format     VARCHAR(16) NOT NULL REFERENCES data_format_versions(format_version),
    supported_features  JSONB,                    -- ["3d_building","hd_road","vector_poi"]
    deprecated_at       DATE
);
```

### compatibility_matrix 表

```sql
CREATE TABLE compatibility_matrix (
    id                  SERIAL PRIMARY KEY,
    engine_version      VARCHAR(16) NOT NULL REFERENCES engine_versions(engine_version),
    format_version      VARCHAR(16) NOT NULL REFERENCES data_format_versions(format_version),
    compatibility_level VARCHAR(16) NOT NULL,     -- FULL / PARTIAL / INCOMPATIBLE
    known_issues        TEXT,
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE (engine_version, format_version)
);
```

### versions 表

```sql
CREATE TABLE versions (
    version_id          BIGSERIAL PRIMARY KEY,
    data_type           data_type_enum NOT NULL,
    version_no          VARCHAR(32) NOT NULL,     -- 如 v20260101
    format_version      VARCHAR(16) NOT NULL REFERENCES data_format_versions(format_version),
    min_engine_version  VARCHAR(16) NOT NULL REFERENCES engine_versions(engine_version),
    base_version_id     BIGINT REFERENCES versions(version_id),  -- 增量基线版本
    status              version_status NOT NULL DEFAULT 'DRAFT',
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    published_at        TIMESTAMPTZ,
    deprecated_at       TIMESTAMPTZ,
    UNIQUE (data_type, version_no)
);

CREATE INDEX idx_versions_status ON versions(status);
CREATE INDEX idx_versions_data_type ON versions(data_type);
```

### version_regions 表

```sql
CREATE TABLE version_regions (
    version_id       BIGINT NOT NULL REFERENCES versions(version_id),
    region_id        VARCHAR(32) NOT NULL REFERENCES regions(region_id),
    status           version_status NOT NULL DEFAULT 'DRAFT',
    tile_count       INTEGER,
    total_size       BIGINT,                     -- 字节数
    checksum_sha256  CHAR(64),
    storage_path     VARCHAR(512),               -- OSS 基础路径
    updated_at       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    PRIMARY KEY (version_id, region_id)
);

CREATE INDEX idx_vr_status ON version_regions(status);
CREATE INDEX idx_vr_region ON version_regions(region_id);
```

### offline_packages 表

```sql
CREATE TABLE offline_packages (
    package_id       BIGSERIAL PRIMARY KEY,
    version_id       BIGINT NOT NULL REFERENCES versions(version_id),
    region_id        VARCHAR(32) NOT NULL REFERENCES regions(region_id),
    package_type     package_type NOT NULL,
    from_version_id  BIGINT REFERENCES versions(version_id),  -- 增量包的基线版本
    package_url      VARCHAR(512) NOT NULL,
    package_size     BIGINT NOT NULL,
    checksum_sha256  CHAR(64) NOT NULL,
    created_at       TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

### tile_index 表

```sql
CREATE TABLE tile_index (
    version_id   BIGINT NOT NULL REFERENCES versions(version_id),
    region_id    VARCHAR(32) NOT NULL REFERENCES regions(region_id),
    data_type    data_type_enum NOT NULL,
    zoom_level   SMALLINT NOT NULL,
    tile_x       INTEGER NOT NULL,
    tile_y       INTEGER NOT NULL,
    storage_key  VARCHAR(512) NOT NULL,          -- OSS 文件路径（CoW 时可指向旧版本路径）
    file_size    BIGINT,
    etag         CHAR(64),
    PRIMARY KEY (version_id, region_id, data_type, zoom_level, tile_x, tile_y)
);

CREATE INDEX idx_tile_index_region ON tile_index(region_id, zoom_level);
```

### version_publish_log 表

```sql
CREATE TABLE version_publish_log (
    log_id      BIGSERIAL PRIMARY KEY,
    version_id  BIGINT NOT NULL REFERENCES versions(version_id),
    region_id   VARCHAR(32) REFERENCES regions(region_id),  -- NULL 表示全局操作
    action      VARCHAR(64) NOT NULL,
    from_status version_status,
    to_status   version_status NOT NULL,
    operator    VARCHAR(128) NOT NULL,
    reason      TEXT,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_publish_log_version ON version_publish_log(version_id);
```

### vehicle_map_status 表

```sql
CREATE TABLE vehicle_map_status (
    vehicle_id       VARCHAR(64) NOT NULL,
    region_id        VARCHAR(32) NOT NULL REFERENCES regions(region_id),
    data_type        data_type_enum NOT NULL,
    engine_version   VARCHAR(16) NOT NULL,
    current_version  VARCHAR(32),               -- 当前使用的 version_no
    update_status    VARCHAR(32),               -- IDLE / DOWNLOADING / PENDING_INSTALL
    last_checked_at  TIMESTAMPTZ,
    updated_at       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    PRIMARY KEY (vehicle_id, region_id, data_type)
);

CREATE INDEX idx_vehicle_status_region ON vehicle_map_status(region_id);
```
